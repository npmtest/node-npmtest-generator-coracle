        formData: {
            'username': config.loginfo.username,
            'pas': config.loginfo.passworld,
            'autologin': ''
        }
    }, function optionalCallback(err, httpResponse, body) {
        if (err) {
            return console.error('failed:', err);
        }
        if (!err && httpResponse.statusCode == 200) {
            JSESSIONID = httpResponse.headers['set-cookie'][1].split(';')[0];
            done();
        }
 
    });
});

gulp.task('upload', ['getFid'], function (done) {
    var cookie = request.cookie(JSESSIONID),
        j = request.jar(),
        count = 0;
    j.setCookie(cookie, config.url);
    for (var i = 0, l = upload_modules.length; i < l; i++) {
        var module = upload_modules[i].module,
            name = upload_modules[i].name;
        var formData = {
            clientVersion: "",
            fid: data[name],
            interfaceUrl: "",
            remark: "",
            workStatus: "usable",
            zipUrl: "",
            uploadFile: {
                value: fs.createReadStream(__dirname + '/' + directory.zip + '/' + module + '.zip'),
                options: {
                    filename: module + '.zip',
                    contentType: 'application/x-zip-compressed'
                }
            }
        };
        request.post({
            url: config.url + '/knd_xSimple/apply/funVer_save.action',
            formData: formData,
            jar: j
        }, function optionalCallback(err, httpResponse, body) {
            if (err) {
                return console.error('failed:', err);
            }
            console.log('successful!' + module + 'has upload');
 
            count += 1;
            if (count === l) {
                done();
            }
        });
    }
});
 
gulp.task('getFid', ['login'], function (done) {

        j = request.jar(),
        count = 0;

            body = JSON.parse(body);

            }
            done();
        }
    });
